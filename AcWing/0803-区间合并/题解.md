
如题所述，这就是个区间合并的模板题。
注意无穷小怎么设置就行。

```
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef pair<int, int> PII;
vector<PII> segs;
const int inf = 0x80000000;

void merge(vector<PII> &segs)
{
    vector<PII> res;
    int st = inf, ed = inf;
    for(auto item:segs)
        if(item.first > ed)
        {
            if(st != inf) res.push_back({st, ed});
            st = item.first, ed = item.second;
        }
        else
            ed = max(ed, item.second);
    
    if(st != inf) res.push_back({st, ed});

    segs = res;
}

int main()
{
    int n;
    scanf("%d", &n);
    
    int l, r;
    for(int i = 0; i < n; i++)
    {
        scanf("%d %d", &l, &r);
        segs.push_back({l, r});
    }
    
    sort(segs.begin(), segs.end());
    
    merge(segs);
    
    printf("%d", segs.size());
    
    return 0;
}
```
